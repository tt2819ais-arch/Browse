name: iOS IPA (unsigned)

on:
  push:
    branches: [ main, master ]
  workflow_dispatch:

jobs:
  build:
    runs-on: macos-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Install XcodeGen
        run: brew install xcodegen

      - name: Generate Xcode project
        run: xcodegen generate

      - name: Detect project & scheme
        id: detect
        run: |
          set -eo pipefail
          PROJ=$(ls -d *.xcodeproj | head -n 1)
          SCHEME=$(xcodebuild -list -project "$PROJ" -json | python3 -c "import json,sys;print(json.load(sys.stdin)['project']['schemes'][0])")
          echo "PROJ=$PROJ" >> "$GITHUB_OUTPUT"
          echo "SCHEME=$SCHEME" >> "$GITHUB_OUTPUT"
          echo "Detected project=$PROJ scheme=$SCHEME"

      - name: Build (unsigned archive)
        run: |
          set -eo pipefail
          xcodebuild \
            -project "${{ steps.detect.outputs.PROJ }}" \
            -scheme "${{ steps.detect.outputs.SCHEME }}" \
            -configuration Release \
            -sdk iphoneos \
            -derivedDataPath build \
            CODE_SIGNING_ALLOWED=NO \
            CODE_SIGNING_REQUIRED=NO \
            CODE_SIGN_IDENTITY="" \
            ONLY_ACTIVE_ARCH=NO \
            clean build

      - name: Package .ipa
        run: |
          set -eo pipefail
          APP_PATH=$(find build/Build/Products/Release-iphoneos -maxdepth 1 -name "*.app" | head -n 1)
          echo "APP_PATH=$APP_PATH"
          if [ -z "$APP_PATH" ]; then echo "No .app produced"; exit 1; fi
          /usr/libexec/PlistBuddy -c "Print :CFBundleExecutable" "$APP_PATH/Info.plist"
          /usr/libexec/PlistBuddy -c "Print :MinimumOSVersion" "$APP_PATH/Info.plist"
          echo "App bundle size:"; du -sh "$APP_PATH"
          rm -rf Payload && mkdir -p Payload
          cp -R "$APP_PATH" Payload/
          zip -qry Aero-unsigned.ipa Payload
          ls -lh Aero-unsigned.ipa

      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: Aero-ipa-unsigned
          path: Aero-unsigned.ipa
          if-no-files-found: error
